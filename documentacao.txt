    private void jButtonCadastrarActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_jButtonCadastrarActionPerformed
        try {

            Usuario user = new Usuario(0, "", jtfNome.getText(), null, jtfValor.getText());
            if (user == null) {
                JOptionPane.showMessageDialog(this, "Valor incorreto. O valor não pode ser negativo.", "Erro", JOptionPane.ERROR_MESSAGE);
                return;
            }
            new UsuarioDao().inserir(user);
            JOptionPane.showMessageDialog(null, "Usuario cadastrado com sucesso");

            setVisible(false);

        } catch (Exception ex) {
            JOptionPane.showMessageDialog(this, "Erro inesperado: " + ex.getMessage(), "Erro", JOptionPane.ERROR_MESSAGE);
        }

    private void jButtonFinalizarEntregaActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_jButtonFinalizarEntregaActionPerformed
        int selectedRow = jTableListagem.getSelectedRow();
        if (selectedRow >= 0) {
            // Pega o valor como String e converte para Integer
            String idString = (String) jTableListagem.getValueAt(selectedRow, 0);
            int idEntrega = Integer.parseInt(idString); // Converte a String para Integer

            try {
                // Atualiza o status da entrega para "Finalizado"
                EntregaDao entregaDao = new EntregaDao();
                entregaDao.alterarStatus(idEntrega, "Finalizado");

                // Atualiza a tabela
                entregas = entregaDao.buscarTodos(); // Recarrega todas as entregas
                preencheTable(entregas);

                JOptionPane.showMessageDialog(null, "Entrega finalizada com sucesso!");
            } catch (ClassNotFoundException | SQLException ex) {
                JOptionPane.showMessageDialog(null, "Erro ao finalizar a entrega: " + ex.getMessage());
            }
        } else {
            JOptionPane.showMessageDialog(null, "Selecione uma entrega para finalizar.");
        }
    }//GEN-LAST:event_jButtonFinalizarEntregaActionPerformed

    private void preencheTable(ArrayList<Entrega> entregas) {
        DefaultTableModel dfte = (DefaultTableModel) jTableListagem.getModel();
        while (dfte.getRowCount() > 0) {
            dfte.removeRow(0);
        }
        for (Entrega e : entregas) {
            String[] linha = {e.getIdEntrega() + "", e.getProduto(), e.getStatus(), e.getDestino()};
            dfte.addRow(linha);

        }
    }

    private void jApagarActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_jApagarActionPerformed
        int row = jTableListagem.getSelectedRow();
        if (row <= -1) {
            JOptionPane.showMessageDialog(null, "selecione uma entrega correta");
        } else {
            // String nome=(String) jTableListagem.getValueAt(row, 1);
            if (JOptionPane.showConfirmDialog(null, "Desaja apagar a Entrega " + entregas.get(row).getProduto() + "?", "Apagar a entrega", JOptionPane.YES_NO_OPTION)
                    == JOptionPane.YES_OPTION) {
                try {
                    new EntregaDao().apagar(entregas.get(row).getIdEntrega());
                    formWindowOpened(null);
                } catch (ClassNotFoundException | SQLException ex) {
                    JOptionPane.showMessageDialog(null, "Erro ao apagar");
                }

            }

        }
    }//GEN-LAST:event_jApagarActionPerformed
